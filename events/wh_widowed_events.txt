# spouse dies, triggers trait assignment. ON DEATH
character_event = {
	id = widowed.3
	picture = GFX_evt_death
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		is_married = yes
		OR = {
			has_character_flag = vlad_von_carstein
			any_spouse = {
				has_character_flag = vlad_von_carstein
			}
			any_spouse = {
				is_alive = yes
				opinion = {
					who = ROOT
					value = 35
				}
				NOT = {
					trait = widowed
				}
			}
		}
	}

	immediate = {
		log = "work event id = widowed.3 for [Root.GetBestName]"
		any_spouse = {
			limit = {
				is_alive = yes
				OR = {
					has_character_flag = vlad_von_carstein
					opinion = {
						who = ROOT
						value = 35
					}
				}
				NOT = {
					trait = widowed
				}
			}
			add_trait = widowed
			character_event = {
				id = widowed.4
				days = 14
			}
		}
	}
}
