namespace = DR_imperial

# THE EMPIRE OF MAN ( or... just the Empire I guess )
# Unification of the Imperial religions
# Sigmarite/Taalite/Ulrican triumphant
narrative_event = {
	id = DR_imperial.1
	title = EVTTITLE_DR_imperial_religion_unification
	desc = EVTDESCA_DR_imperial_religion_unification
	picture = GFX_evt_icon
	is_triggered_only = yes

	immediate = {
		log = "work event DR_imperial.1"
	}

	option = {
		name = EVTOPTA_DR_imperial_religion_unification
		IF = {
			limit = {
				religion = sigmarite
			}
			set_parent_religion = {
				religion = ulrican parent = sigmarite
			}
			set_parent_religion = {
				religion = taalite parent = sigmarite
			}
			give_nickname = nick_the_pious
		}
		IF = {
			limit = {
				religion = ulrican
			}
			set_parent_religion = {
				religion = sigmarite parent = ulrican
			}
			set_parent_religion = {
				religion = taalite parent = ulrican
			}
		}
		IF = {
			limit = {
				religion = taalite
			}
			set_parent_religion = {
				religion = sigmarite parent = taalite
			}
			set_parent_religion = {
				religion = ulrican parent = taalite
			}
		}
		set_global_flag = imperial_churches_unified
		hidden_tooltip = {
			IF = {
				limit = {
					religion = sigmarite
				}
				any_playable_ruler = {
					limit = {
						OR = {
							religion = ulrican
							religion = taalite
						}
						controls_religion = no
					}
					narrative_event = {
						id = DR_imperial.2
					}
				}
			}
			IF = {
				limit = {
					religion = ulrican
				}
				any_playable_ruler = {
					limit = {
						OR = {
							religion = sigmarite
							religion = taalite
						}
						controls_religion = no
					}
					narrative_event = {
						id = DR_imperial.2
					}
				}
			}
			IF = {
				limit = {
					religion = taalite
				}
				any_playable_ruler = {
					limit = {
						OR = {
							religion = ulrican
							religion = sigmarite
						}
						controls_religion = no
					}
					narrative_event = {
						id = DR_imperial.2
					}
				}
			}
			religion_authority = {
				modifier = faith_triumphant
			}
		}
	}
}

# Convert or Not?
narrative_event = {
	id = DR_imperial.2
	title = EVTTITLE_DR_imperial_religion_unification_conversion
	desc = EVTDESC_DR_imperial_religion_unification_conversion
	picture = GFX_evt_icon
	is_triggered_only = yes

	immediate = {
		log = "work event DR_imperial.2"
	}

	option = {
		# Yes
		name = EVTOPTA_DR_imperial_religion_unification
		ai_chance = {
			factor = 50
			modifier = {
				factor = 5
				any_liege = {
					religion = FROM
				}
			}
			modifier = {
				factor = 5
				trait = cynical
			}
			modifier = {
				factor = 10
				dynasty = FROM
			}
			modifier = {
				factor = 0.1
				trait = zealous
			}
			modifier = {
				factor = 0.5
				trait = content
			}
		}
		religion = FROM
		any_courtier = {
			religion = FROM
		}
		capital_scope = {
			religion = FROM
		}
	}

	option = {
		# No
		name = EVTOPTBLoR.211
		tooltip = {
			religion = THIS
		}
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				trait = content
			}
			modifier = {
				factor = 5
				trait = zealous
			}
			modifier = {
				# Ulricans in Middenland are specially stubborn
				factor = 8
				religion = ulrican
				capital_scope = {
					kingdom = {
						title = k_middenland
					}
				}
			}
		}
	}
}

# Arch-Lector elector title "inheritance" ( on_death )
narrative_event = {
	id = DR_imperial.3
	# title = EVTTITLE_DR_imperial_religion_unification_conversion
	# desc = EVTDESC_DR_imperial_religion_unification_conversion
	# picture = GFX_evt_icon
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			has_landed_title = d_arch_lectory_talabheim
			has_landed_title = d_arch_lectory_nuln
		}
	}

	immediate = {
		log = "work event DR_imperial.3"
		if = {
			limit = {
				ROOT = {
					has_landed_title = d_arch_lectory_talabheim
				}
				d_arch_lectory_talabheim = {
					holder_scope = {
						piety = 500 # So instead, give some piety to the Arch-Lectors so that they're elected as electors ( blergh )
					}
				}
			}
		}
		if = {
			limit = {
				ROOT = {
					has_landed_title = d_arch_lectory_nuln
				}
				d_arch_lectory_nuln = {
					holder_scope = {
						piety = 500 # So instead, give some piety to the Arch-Lectors so that they're elected as electors ( blergh )
					}
				}
			}
		}
		# current_heir = {
		# # give_religious_title = title_lector # ... there's no effect to give religious titles, great
		# }
	}

	option = {
		# Yes
		name = OK
	}
}

# If there aren't any Arch-Lectors, give 2 Lectors the title
character_event = {
	id = DR_imperial.7
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		log = "work event DR_imperial.7"
		IF = {
			limit = {
				d_arch_lectory_talabheim = {
					holder_scope = {
						OR = {
							controls_religion = yes
							is_alive = no
						}
					}
				}
			}
			d_arch_lectory_talabheim = {
				grant_title = ROOT
			}
			b_vateresche = {
				grant_title = ROOT
			}
			break = yes
		}
		IF = {
			limit = {
				d_arch_lectory_nuln = {
					holder_scope = {
						OR = {
							controls_religion = yes
							is_alive = no
						}
					}
				}
			}
			d_arch_lectory_nuln = {
				grant_title = ROOT
			}
			b_tempel_quarter = {
				grant_title = ROOT
			}
			break = yes
		}
	}

	option = {
		name = ok
	}
}
