# Become friendly with a fellow vassal when both at same location and we got same liege
character_event = {
	id = 6900
	desc = EVTDESC6900
	picture = GFX_evt_battle
	only_men = yes
	min_age = 16
	capable_only = yes
	only_rulers = yes
	prisoner = no
	is_triggered_only = yes

	trigger = {
		in_command = yes
		independent = no
		NOT = {
			trait = shy
		}
		liege = {
			any_vassal = {
				NOT = {
					character = ROOT
				}
				in_command = yes
				at_location = ROOT
				nor = {
					war_with = ROOT
					trait = shy
				}
				opinion = {
					who = ROOT value = -30
				}
				NOT = {
					opinion = {
						who = ROOT value = 30
					}
				}
			}
		}
	}

	weight_multiplier = {
		factor = 0.10000
		modifier = {
			factor = 1.25
			trait = gregarious
		}
		modifier = {
			factor = 1.25
			trait = brave
		}
	}

	immediate = {
		log = "work event id = 6900 for [Root.GetBestName]"
		liege = {
			random_vassal = {
				limit = {
					NOT = {
						character = ROOT
					}
					in_command = yes
					at_location = ROOT
					nor = {
						war_with = ROOT
						trait = shy
					}
					opinion = {
						who = ROOT value = -30
					}
					NOT = {
						opinion = {
							who = ROOT value = 30
						}
					}
				}
				save_event_target_as = campaign_friend_6900
			}
		}
	}

	option = {
		name = EVTOPTA6900
		event_target:campaign_friend_6900 = {
			opinion = {
				modifier = campaign_friend
				who = ROOT
				years = 10
			}
			hidden_tooltip = {
				character_event = {
					id = 6901 days = 1
				}
			}
		}
	}
}

# Save someone's life at the battlefield... become very good friend.

	only_men = yes
	capable_only = yes
	only_rulers = yes
	prisoner = no
	is_triggered_only = yes

		in_command = yes
		in_battle = yes
		independent = no
		nor = {
			trait = craven
			health_traits = 1
		}
		liege = {
			any_vassal = {
				NOT = {
					character = ROOT
				}
				in_command = yes
				in_battle = yes
				at_location = ROOT
				nor = {
					war_with = ROOT
					health_traits = 1
				}
				opinion = {
					who = ROOT value = -30
				}
				NOT = {
					opinion = {
						who = ROOT value = 30
					}
				}
			}
		}
	}

	weight_multiplier = {
		factor = 0.10000
		modifier = {
			factor = 2.00
			trait = brave
		}
	}

	immediate = {
		log = "work event id = 6910 for [Root.GetBestName]"
		liege = {
			random_vassal = {
				limit = {
					NOT = {
						character = ROOT
					}
					in_command = yes
					at_location = ROOT
					in_battle = yes
					nor = {
						health_traits = 1
						war_with = ROOT
					}
					opinion = {
						who = ROOT value = -30
					}
					NOT = {
						opinion = {
							who = ROOT value = 30
						}
					}
				}
				save_event_target_as = campaign_friend_6910
			}
		}
	}

	option = {
		name = EVTOPTA6910
		event_target:campaign_friend_6910 = {
			prestige = 5
			add_friend = ROOT
			if = {
				limit = {
					has_dlc = "Zeus"
				}
				reverse_add_favor = ROOT
			}
			hidden_tooltip = {
				character_event = {
					id = 6911 days = 1
				}
			}
		}
	}
}
