# JOINING
# From on_character_ask_to_join_society
character_event = {
	id = DR_miners_guild.100
	hide_window = yes
	is_triggered_only = yes

	trigger = {
 		FROM = {
			leader = {
				society_member_of = miners_guild
			}
		}
	}

	immediate = {
		log = "work event id = DR_miners_guild.100 for [Root.GetBestName]"
 		FROM = {

				character_event = {
					id = DR_miners_guild.101
				}
			}
		}
	}
}
