# If in the red = > increase revolt risk in a province.
province_event = {
	id = 68000
	desc = "EVTDESC68000"
	picture = "GFX_evt_bandits"
	notification = yes
	is_triggered_only = yes

	trigger = {
		NOT = {
			owner = {
				wealth = 0
			}
		}
		NOT = {
			has_province_modifier = incompetent_rule
		}
	}

	weight_multiplier = {
		factor = 0.50000
		# lower if really bad demesne efficiency
		modifier = {
			factor = 1.11
			NOT = {
				owner = {
					wealth = -50
				}
			}
		}
		modifier = {
			factor = 1.43
			NOT = {
				owner = {
					wealth = -100
				}
			}
		}
		modifier = {
			factor = 2.00
			NOT = {
				owner = {
					wealth = -200
				}
			}
		}
		modifier = {
			factor = 3.33
			NOT = {
				owner = {
					wealth = -300
				}
			}
		}
		modifier = {
			factor = 10.00
			NOT = {
				owner = {
					wealth = -500
				}
			}
		}
		# traits affect this
		modifier = {
			factor = 0.67
			owner = {
				trait = ambitious
			}
		}
		modifier = {
			factor = 0.20
			owner = {
				trait = just
			}
		}
		modifier = {
			factor = 2.00
			owner = {
				trait = arbitrary
			}
		}
		modifier = {
			factor = 1.43
			owner = {
				trait = greedy
			}
		}
		modifier = {
			factor = 1.43
			owner = {
				trait = cruel
			}
		}
		modifier = {
			factor = 10.00
			owner = {
				trait = slothful
			}
		}
		modifier = {
			factor = 5.00
			owner = {
				trait = incapable
			}
		}
		modifier = {
			factor = 10.00
			owner = {
				trait = infirm
			}
		}
		# modify by stewardship and intrigue
		modifier = {
			factor = 1.43
			NOT = {
				owner = {
					realm_stewardship = 15
				}
			}
		}
		modifier = {
			factor = 1.43
			NOT = {
				owner = {
					realm_stewardship = 10
				}
			}
		}
		modifier = {
			factor = 1.43
			NOT = {
				owner = {
					realm_stewardship = 5
				}
			}
		}
		modifier = {
			factor = 0.50
			owner = {
				realm_stewardship = 25
			}
		}
		modifier = {
			factor = 0.67
			owner = {
				realm_stewardship = 20
			}
		}
		modifier = {
			factor = 0.50
			owner = {
				realm_intrigue = 25
			}
		}
		modifier = {
			factor = 0.67
			owner = {
				realm_intrigue = 20
			}
		}
	}

	immediate = {
		log = "work event id = 68000 for [Root.GetBestName]"
		add_province_modifier = {
			name = incompetent_rule
			duration = -1
		}
	}

	option = {
		name = "EVTOPTA68000"
	}
}
