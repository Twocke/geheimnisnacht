namespace = stealing

# stealing decision is triggered
character_event = {
	id = stealing.0
	title = THEFT # Theft
	desc = "EVTDESCSTEALING.0"
	picture = "GFX_evt_into_the_dungeon"
	is_triggered_only = yes

	immediate = {
		log = "work event id = stealing.0 for [Root.GetBestName]"
	}

	option = {
		name = OF_COURSE
		if = {
			limit = {
				NOT = {
					trait = greedy
				}
			}
			random = {
				chance = 30
				add_trait = greedy
			}
		}
		prestige = -200
		FROM = {
			character_event = {
				id = stealing.1
			}
		}
	}

	option = {
		name = NO
	}
}

# prisoner is informed
character_event = {
	id = stealing.1
	title = THEFT # Theft
	desc = "EVTDESCSTEALING.1"
	picture = "GFX_evt_into_the_dungeon"
	is_triggered_only = yes

	immediate = {
		log = "work event id = stealing.1 for [Root.GetBestName]"
	}

	option = {
		name = "BASTARDS"
		any_artifact = {
			limit = {
				nor = {
					has_artifact_flag = mount
					has_artifact_flag = carstein_ring
				}
			}
			transfer_artifact = {
				from = root
				to = from
			}
		}
	}
}
